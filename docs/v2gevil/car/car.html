 28    def __init__(
 29        self,
 30        interface: str = "eth_car",
 31        ipv6_address: str = V2GTPAddress.CAR.value,
 32        protocol: bytes = V2GTPProtocols.TCP.value,
 33        tls_flag: bool = False,
 34        accept_security: bool = True,
 35    ):
 36        """Initialize."""
 37        self.interface = interface
 38        self.ipv6_address = ipv6_address
 39        self.protocol = protocol
 40        self.tls_flag = tls_flag
 41        self.accept_security = accept_security
 42        self.tcp_server_address = None
 43        self.tcp_server_port = None
 44
 45    def sdp_client(self):
 46        """Start SDP client."""
 47
 48        # SECC Discovery Protocol Server port
 49        port = V2G_UDP_SDP_SERVER
 50
 51        # Create a UDP socket
 52        SDP_socket = socket.socket(
 53            family=socket.AF_INET6, type=socket.SOCK_DGRAM
 54        )
 55
 56        # IPv6 multicast group address for all nodes on the local network segment
 57        # Check with %eth_car, seems to work both ways
 58        # Have to have .value, because otherwise it's not a string type, but Enum type
 59        # str type is needed for socket.getaddrinfo(), otherwise it will raise Error
 60        multicast_address = V2GTPAddress.MULTICAST_ADDRESS.value
 61        link_local_address = V2GTPAddress.CAR.value
 62
 63        # Get the interface index of the interface with the specified name
 64        interface_index = socket.if_nametoindex(self.interface)
 65
 66        src_port = random.choice(V2G_UDP_SDP_CLIENT)
 67        SDP_socket.bind((link_local_address, src_port, 0, interface_index))
 68        SDP_socket.setsockopt(
 69            socket.IPPROTO_IPV6,
 70            socket.IPV6_MULTICAST_IF,
 71            struct.pack("I", interface_index),
 72        )
 73
 74        sock_addr = socket.getaddrinfo(
 75            multicast_address, port, socket.AF_INET6, socket.SOCK_DGRAM
 76        )[0][4]
 77        print(f"Send to:{sock_addr}")
 78
 79        msg = V2GTPMessage()
 80        sdp_request = msg.create_v2gtp_sdp_request(
 81            protocol=self.protocol, tls_flag=self.tls_flag
 82        )
 83        SDP_socket.sendto(sdp_request, sock_addr)
 84        print(SDP_socket.getsockname())
 85
 86        # Sending request until response is received
 87        while True:
 88            data, server_address = SDP_socket.recvfrom(1024)
 89            if data:
 90                security_byte, protocol_byte = self.sdp_response_handler(
 91                    data, server_address
 92                )
 93                # If True, TLS will be used if is in SDP response from server
 94                if self.accept_security:
 95                    if security_byte == V2GTPProtocols.TLS:
 96                        self.tls_flag = True
 97                    elif security_byte == V2GTPProtocols.NO_TLS:
 98                        self.tls_flag = False
 99                # False => security byte has to be the same as in SDP request
100                # If security byte is different, raise ValueError
101                else:
102                    if security_byte == V2GTPProtocols.TLS and self.tls_flag:
103                        pass
104                    elif (
105                        security_byte == V2GTPProtocols.NO_TLS
106                        and not self.tls_flag
107                    ):
108                        pass
109                    else:
110                        raise ValueError(
111                            "TLS flag in SDP response is different"
112                            "than in SDP request and user chose not to accept"
113                        )
114                break
115
116        SDP_socket.close()
117        print("SDP client finished")
118        return security_byte, protocol_byte

120    def sdp_response_handler(self, data: bytes, server_address: str):
121        """Handle SDP response from SDP server."""
122        sdp_response = V2GTPMessage(data)
123        print(f"Received response: {data} from {server_address}")
124        (
125            self.tcp_server_address,
126            self.tcp_server_port,
127            security_byte,
128            protocol_byte,
129        ) = sdp_response.parse_v2gtp_sdp_response()
130
131        # Convert from bytes to int
132        self.tcp_server_port = int.from_bytes(
133            self.tcp_server_port, byteorder="big"
134        )
135        # Convert from packet binary format to IPv6 address string format
136        self.tcp_server_address = socket.inet_ntop(
137            socket.AF_INET6, self.tcp_server_address
138        )
139
140        print(
141            f"Server address: {self.tcp_server_address}, "
142            f"server port: {self.tcp_server_port}, "
143            f"security byte: {security_byte}, "
144            f"protocol byte: {protocol_byte}\n"
145        )
146
147        return security_byte, protocol_byte

149    def tcp_client(self, security_byte: bytes):
150        """Start TCP client."""
151
152        print("TCP client started")
153        interface_index = socket.if_nametoindex(self.interface)
154        # TCP server port and address will be received from SDP response
155        # For now, it's hardcoded
156        if security_byte == V2GTPProtocols.NO_TLS:
157            # TODO: Run on plain TCP
158            link_local_address = self.ipv6_address
159            src_port = random.choice(
160                range(
161                    V2GTPPorts.V2G_SRC_TCP_DATA_START.value,
162                    V2GTPPorts.V2G_SRC_TCP_DATA_END.value,
163                )
164            )
165
166            with socket.socket(socket.AF_INET6, socket.SOCK_STREAM) as sock:
167                sock.bind((link_local_address, src_port, 0, interface_index))
168                sock.connect(
169                    (self.tcp_server_address, self.tcp_server_port, 0, 0)
170                )
171                sock.sendall(b"TCP Hello from client")
172                data = sock.recv(1024)
173                print(f"Received {data}")
174
175            sock.close()
176
177            # print(f"Received {data.decode('utf-8')}")
178        elif security_byte == V2GTPProtocols.TLS:
179            # TODO: Add TLS support, TLS server, it's not implemented yet
180            raise NotImplementedError("TLS server is not implemented yet")
181        else:
182            raise ValueError("Unknown security byte")

233def start_sdp_client(
234    interface: str = "eth_car",

238    """Start SDP client."""

240    port = V2G_UDP_SDP_SERVER
241
242    # Create a UDP socket
243    SDP_socket = socket.socket(family=socket.AF_INET6, type=socket.SOCK_DGRAM)
244
245    # IPv6 multicast group address for all nodes on the local network segment
246    # Check with %eth_car, seems to work both ways
247    multicast_address = "ff02::1"
248    link_local_address = "fe80::d237:45ff:fe88:b12a"
249
250    # Get the interface index of the interface with the specified name
251    interface_index = socket.if_nametoindex(interface)
252    # Port number 0 => let the system choose a free port
253    # flowinfo is set to 0 also
254    # Bind for the link-local address is necessary, because the SDP server
255    # sends the response to the link-local address of the client
256    src_port = random.choice(V2G_UDP_SDP_CLIENT)
257    SDP_socket.bind((link_local_address, src_port, 0, interface_index))
258
259    # Set the interface for sending multicast packets
260    # Seems to be not necessary if SDP_socket.bind() is used
261    # I use also bind() because of the setting port number and link-local address
262    # Withou bind():
263    # getsockname() ('::', 48982, 0, 0)
264    # with bind():
265    # getsockname() ('fe80::d237:45ff:fe88:b12a', 48982, 0, 16)
266    # If setsockopt() or bind() isn't used =>
267    # the socket is not bound to the interface
268    # TODO: Add bind because of the setting port number
269    SDP_socket.setsockopt(
270        socket.IPPROTO_IPV6,
271        socket.IPV6_MULTICAST_IF,
272        struct.pack("I", interface_index),
273    )
274
275    sock_addr = socket.getaddrinfo(
276        multicast_address, port, socket.AF_INET6, socket.SOCK_DGRAM
277    )[0][4]
278    print(f"Send to:{sock_addr}")
279
280    # TODO: Send regular V2G SDP request message
281    msg = V2GTPMessage()
282    sdp_request = msg.create_v2gtp_sdp_request(
283        protocol=protocol, tls_flag=tls_flag
284    )
285    SDP_socket.sendto(sdp_request, sock_addr)
286    print(SDP_socket.getsockname())
287
288    # Sending request until response is received
289    while True:
290        data, server_address = SDP_socket.recvfrom(1024)
291        if data:
292            # TODO: Use server_address and port in start_tcp_client method
293            sdp_response = V2GTPMessage(data)
294            print(f"Received response: {data} from {server_address}")
295            (
296                server_address,
297                server_port,
298                security_byte,
299                protocol_byte,
300            ) = sdp_response.parse_v2gtp_sdp_response()
301            print(
302                f"Server address: {server_address}, "
303                f"server port: {server_port}, "
304                f"security byte: {security_byte}, "
305                f"protocol: {protocol_byte}"
306            )
307            break
308
309    SDP_socket.close()
310    print("SDP client finished")
