v2gevil.messages.generator

This module contains the message generator for the V2G communication.

The message generator is responsible for generating the dictionary, in which is name of the request message and the corresponding response message as key-value pair. The value is a dictionary, which contains the name and value for each field of the response message.

 114    def add_model_to_dict(
 115        self,
 116        dictionary: dict,
 117        model_obj: Union[BodyBaseType, supportedAppProtocolRes],
 118    ) -> dict:
 119        """
 120        Add model_dump to dictionary.
 121        Key is the name of the class of the model object.
 122        Value is the model_dump of the model object.
 123        """
 124
 125        dict_obj = model_obj.model_dump(by_alias=True, exclude_unset=True)
 126        response_msg_name = model_obj.__class__.__name__
 127        request_msg_name = model_obj.request_class_name()
 128
 129        dictionary[request_msg_name] = {response_msg_name: dict_obj}
 130
 131        logger.debug(dict_obj)
 132        logger.debug(dictionary)
 133
 134        return dictionary

 136    def load_default_dict(self, file_name: EVSEDefaultDictPaths) -> dict:
 137        """Load default dictionary from file."""
 138
 139        msg_abs_path = Path(__file__).parent.absolute()
 140        file_name_path = msg_abs_path.joinpath(file_name.value)
 141        logger.debug(
 142            "Trying to load default dictionary from file: %s", file_name_path
 143        )
 144
 145        if Path(file_name_path).is_file():
 146            logger.info(
 147                "Default dictionary already exists."
 148                "If you want to override it, set override_flag to True."
 149            )
 150            # Load file and return dict loaded from file
 151            with open(
 152                file_name_path, "r", encoding="utf-8"
 153            ) as dictionary_file:
 154                dictionary = json.load(dictionary_file)
 155                logger.debug(type(dictionary))
 156                logger.debug(dictionary)
 157
 158            return dictionary
 159
 160        return {}

 176    def generate_default_dict_AC(self, override_flag: bool = False) -> dict:
 177        """Generate default dictionary for AC charging mode.
 178        
 179        Generated dictionary is also saved to file\
 180            if not exists or if override_flag is True.
 181                    
 182        Args:
 183            override_flag (bool, optional): If True, existing file will be overriden.\
 184                Defaults to False.
 185
 186        Returns:
 187            Generated dictionary for AC charging mode.
 188        """
 189
 190        default_dict_ac = {}
 191
 192        if not override_flag:
 193            default_dict_ac = self.load_default_dict(
 194                EVSEDefaultDictPaths.AC_MODE_PATH
 195            )
 196            # Dict is not empty
 197            if default_dict_ac:
 198                return default_dict_ac
 199
 200        # Override_flag is True or default dictionary is empty
 201
 202        generator_functions = [
 203            # Common AC/DC messages START
 204            self.gen_default_supportedAppProtocolRes,
 205            self.gen_default_SessionSetupRes,
 206            self.gen_default_ServiceDiscoveryRes,
 207            self.gen_default_ServiceDetailRes,
 208            self.gen_default_PaymentServiceSelectionRes,
 209            self.gen_default_PaymentDetailsRes,
 210            self.gen_default_AuthorizationRes,
 211            self.gen_default_ChargeParameterDiscoveryRes,
 212            self.gen_default_PowerDeliveryRes,
 213            self.gen_default_MeteringReceiptRes,
 214            self.gen_default_SessionStopRes,
 215            # TODO: Implement following
 216            self.gen_default_CertificateUpdateRes,
 217            self.gen_default_CertificateInstallationRes,
 218            # Common AC/DC messages END
 219            # AC messages START
 220            self.gen_default_ChargingStatusRes,
 221            # AC messages END
 222        ]
 223
 224        # Call all generator functions and add the generated response message
 225        # add the generated response message to the dictionary
 226        # self.charging_mode is EVSEChargingMode.AC
 227        # So there is no need to explicitly call methods with charging_mode=AC
 228        # All method should check if charging_mode in argument is None
 229        # and if None it will use the self.charging_mode
 230        for gen_func in generator_functions:
 231            obj = gen_func()
 232            default_dict_ac = self.add_model_to_dict(default_dict_ac, obj)
 233
 234        # Save dictionary to file
 235        self.save_default_dict(
 236            EVSEDefaultDictPaths.AC_MODE_PATH, default_dict_ac
 237        )
 238        logger.debug(
 239            "Default dictionary for AC mode is generated: %s", default_dict_ac
 240        )
 241        return default_dict_ac

 305    def generate_default_dict(self, override_flag: bool = False) -> dict:
 306        """Generate default dictionary for all messages.
 307        
 308        Save generated default dictionary into the file\
 309            if not exists or if override_flag is True
 310
 311        Args:
 312            override_flag (bool, optional): If True, existing file will be overriden.\
 313                Defaults to False.
 314
 315        Returns:
 316            Generated dictionary for AC or DC charging mode, based on the\
 317                charging mode of the class.
 318        """
 319
 320        if self.charging_mode == EVSEChargingMode.AC:
 321            return self.generate_default_dict_AC(override_flag)
 322        if self.charging_mode == EVSEChargingMode.DC:
 323            return self.generate_default_dict_DC(override_flag)
 324
 325        # Charging mode is not supported
 326        logger.error("Charging mode %s is not supported.", self.charging_mode)
 327        logger.error("Default dictionary can not be generated.")
 328        raise ValueError(
 329            f"Charging mode {self.charging_mode} is not supported."
 330        )
