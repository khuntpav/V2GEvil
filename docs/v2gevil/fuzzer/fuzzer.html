 90    def fuzz(self, message_name: str = ""):
 91        """Fuzz EVSE response messages."""
 92        # Check which fuzzing mode is selected
 93        # and call appropriate method
 94        match self.mode:
 95            # Fuzz all messages and all possible parameters in messages
 96            # Fuzzing all, config file or custom dict is NOT used
 97            # all parameters will be fuzzed, values are random choice
 98            case EVFuzzMode.ALL:
 99                self.fuzz_all()
100            # For fuzzing use values from custom dictionary
101            # This depends only on the provided custom dictionary
102            # config file is NOT used
103            case EVFuzzMode.CUSTOM:
104                self.fuzz_custom()
105            # Fuzz only one message, specified by message_name
106            # parameters to be fuzzed are chosen based on fuzzer config file
107            # This is a combination of user choice and fuzzer config file
108            # custom dict is NOT used
109            case EVFuzzMode.MESSAGE:
110                if message_name == "":
111                    logger.error(
112                        "Message name is not set. "
113                        "Cannot fuzz only one message."
114                    )
115                    raise ValueError(
116                        "Message name is not set. "
117                        "Cannot fuzz only one message."
118                    )
119                self.fuzz_message(message_name=message_name)
120            # Which messages and which parameters will be fuzzed depends
121            # only on the fuzzer config file
122            # custom dict is NOT used
123            case EVFuzzMode.CONFIG:
124                self.fuzz_config_based()
125            case _:
126                logger.error("Invalid fuzzing mode: %s", self.mode)
127                raise ValueError(f"Invalid fuzzing mode: {self.mode}")
128
129        # In the end station is started with fuzzing_dict instead of default_dict
130        # validate=False because validation is not desirable, because fuzzing
131        # can produce invalid types or values for parameters of the messages
132        print(80 * "*")
133        station.start_async(
134            interface=self.interface,
135            charging_mode=self.charging_mode,
136            req_res_map=self.fuzzing_dict,
137            validate=False,
138        )

188    def fuzz_message(self, message_name: str):
189        """Fuzz only specified message in fuzzing_dict. Message is specified by message_name.
190
191        Args:
192            message_name (str): Name of message to fuzz.
193        """
194
195        # Need conversion to enum, because user input is string
196        try:
197            message_name = MessageName(message_name)
198        except ValueError as exc:
199            logger.error("Invalid message name: %s", message_name)
200            raise ValueError(f"Invalid message name: {message_name}") from exc
201
202        # Load fuzzer config file in binary mode
203        # rb need to be used because of tomli library
204        with open(self.config_filename, "rb") as config_file:
205            config_data = tomli.load(config_file)
206
207        # Check if message name is in config file
208        if message_name not in config_data.keys():
209            logger.error(
210                "Message name: %s is not in fuzzer config file.", message_name
211            )
212            logger.error("Fuzzer config file: %s", self.config_filename)
213            raise ValueError(
214                f"Message name: {message_name} is not in fuzzer config file"
215            )
216
217        # Pick only config for specified message
218        msg_config = config_data[message_name]
219
220        # Differ in "match case", which fuzz method will be called
221        # based on the message name
222        match message_name:
223            case MessageName.SUPPORTED_APP_PROTOCOL_RES:
224                self.fuzz_supported_app_protocol_res(msg_config=msg_config)
225            case MessageName.SESSION_SETUP_RES:
226                self.fuzz_session_setup_res(msg_config=msg_config)
227            case MessageName.SERVICE_DISCOVERY_RES:
228                self.fuzz_service_discovery_res(msg_config=msg_config)
229            case MessageName.SERVICE_DETAIL_RES:
230                self.fuzz_service_detail_res(msg_config=msg_config)
231            case MessageName.PAYMENT_SERVICE_SELECTION_RES:
232                self.fuzz_payment_service_selection_res(msg_config=msg_config)
233            case MessageName.PAYMENT_DETAILS_RES:
234                self.fuzz_payment_details_res(msg_config=msg_config)
235            case MessageName.AUTHORIZATION_RES:
236                self.fuzz_authorization_res(msg_config=msg_config)
237            case MessageName.CHARGE_PARAMETER_DISCOVERY_RES:
238                self.fuzz_charge_parameter_discovery_res(msg_config=msg_config)
239            case MessageName.POWER_DELIVERY_RES:
240                self.fuzz_power_delivery_res(msg_config=msg_config)
241            case MessageName.METERING_RECEIPT_RES:
242                self.fuzz_metering_receipt_res(msg_config=msg_config)

261                raise ValueError(f"Invalid message name: {message_name}")

263    def fuzz_config_based(self):
264        """Fuzz messages and parameters in fuzzing_dict based on config file."""
265        # Load config file in binary mode
266        # rb need to be used because of tomli library
267        with open(self.config_filename, "rb") as config_file:
268            config_data = tomli.load(config_file)
269
270        # Get all messages names from config file, which should be fuzzed
271        message_names = config_data.keys()
272
273        # Pairs message name and fuzzing method
274        pairs_message_fuzz_method = {
275            MessageName.SUPPORTED_APP_PROTOCOL_RES: self.fuzz_supported_app_protocol_res,
276            MessageName.SESSION_SETUP_RES: self.fuzz_session_setup_res,
277            MessageName.SERVICE_DISCOVERY_RES: self.fuzz_service_discovery_res,
278            MessageName.SERVICE_DETAIL_RES: self.fuzz_service_detail_res,
279            MessageName.PAYMENT_SERVICE_SELECTION_RES: self.fuzz_payment_service_selection_res,
280            MessageName.PAYMENT_DETAILS_RES: self.fuzz_payment_details_res,
281            MessageName.AUTHORIZATION_RES: self.fuzz_authorization_res,
282            MessageName.CHARGE_PARAMETER_DISCOVERY_RES: self.fuzz_charge_parameter_discovery_res,
283            MessageName.POWER_DELIVERY_RES: self.fuzz_power_delivery_res,
284            MessageName.METERING_RECEIPT_RES: self.fuzz_metering_receipt_res,
285            MessageName.SESSION_STOP_RES: self.fuzz_session_stop_res,
286            MessageName.CERTIFICATE_UPDATE_RES: self.fuzz_certificate_update_res,
287            MessageName.CERTIFICATE_INSTALLATION_RES: self.fuzz_certificate_installation_res,
288            MessageName.CHARGING_STATUS_RES: self.fuzz_charging_status_res,
289            MessageName.CABLE_CHECK_RES: self.fuzz_cable_check_res,
290            MessageName.PRE_CHARGE_RES: self.fuzz_pre_charge_res,
291            MessageName.CURRENT_DEMAND_RES: self.fuzz_current_demand_res,
292            MessageName.WELDING_DETECTION_RES: self.fuzz_welding_detection_res,
293        }
294
295        # Iterate over all message names in config file and
296        # call appropriate fuzzing method
297        for name in message_names:
298            # Check if message name is valid
299            try:
300                name = MessageName(name)
301            except ValueError as exc:
302                logger.error("Invalid message name: %s in fuzzer config", name)
303                raise ValueError(f"Invalid message name: {name}") from exc
304            # Call appropriate fuzzing method
305            pairs_message_fuzz_method[name](msg_config=config_data[name])

309    def fuzz_supported_app_protocol_res(
310        self, msg_config: Optional[dict] = None
311    ):
312        """Fuzz supportedAppProtocolRes message in fuzzing_dict
313
314        Args:
315            params (list, optional): List of parameters to fuzz. Defaults to None.
316            If None, all parameters will be fuzzed.
317                Possible values: SchemaID, ResponseCode
318        """
319        # self.fuzzing_dict contains all messages
320        # pair req and response message {req_name: {res_name: {key: value, key: value, ...}}}
321        # => {supportedAppProtocolReq: {supportedAppProtocolRes: {key: value, key: value, ...}}}
322        # key is req and value is response dict
323        # response dict is {res_name: {key: value, key: value, ...}}
324        # It's like this because in create response is also need to know response name
325        # because it's needed when Body.model_validate is called
326        # => the name of Body attribute is the same as response name
327        # but here i need to know only response dict => that's why [req_name][res_name]
328
329        # MODE ALL - all parameters will be fuzzed - msg_config is None
330        # Config file is NOT used
331        # msg_config is not None for modes: MESSAGE, CONFIG
332        # Mode MESSAGE - fuzz only one message params specified in config file
333        # Mode CONFIG - fuzz all messages and params specified in config file
334
335        # In dict works without .value
336        req_key = MessageName.SUPPORTED_APP_PROTOCOL_REQ
337        res_key = MessageName.SUPPORTED_APP_PROTOCOL_RES
338
339        msg_fuzz_dict = self.fuzzing_dict[req_key][res_key]
340        # Keep default values for all params in the message
341        msg_default_dict = self.default_dict[req_key][res_key]
342
343        msg_fuzzer = FuzzerSupportedAppProtocolRes(
344            msg_config=msg_config,
345            msg_fuzz_dict=msg_fuzz_dict,
346            msg_default_dict=msg_default_dict,
347        )
348
349        # Replace message in fuzzing_dict with fuzzed one (msg_dict_to_fuzz)
350        self.fuzzing_dict[req_key][res_key] = msg_fuzzer.fuzz()

273
274        # Pairs message name and fuzzing method
275        pairs_message_fuzz_method = {
276            MessageName.SUPPORTED_APP_PROTOCOL_RES: self.fuzz_supported_app_protocol_res,
