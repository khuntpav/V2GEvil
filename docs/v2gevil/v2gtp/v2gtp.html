 66        inverse_version = V2GTPVersion.CURRENT_VERSION_INVERSE

551    """Check if payload type in header is V2GTP EXI message."""
552
553    header, payload = parse_v2gtp_pkt(pkt)
554    if header is None or payload is None:
555        return False
556    # From byte 3 to byte 4 is payload type [2:4] from 2. position to 3. position
557    # 4. position is not included, so it's byte 3 and 4
558    # Bytes in ISO, number of byte starts from 1, not from 0
559    # So byte 3 and 4 is here 2. and 3. position
560    payload_type = header[2:4]
561    if payload_type == b"\x80\x01":
562        return True
563
564    return False
