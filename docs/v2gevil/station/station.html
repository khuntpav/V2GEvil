164    async def sdp_server(self):
165        """Start SDP server.
166
167        The SDP server is started on UDP (multicast) port 15118 (defined in ISO15118-2).
168        Should accepts UDP packets with a local-link IP multicast destination address
169        """
170        print("SDP server started")
171
172        # Create a UDP socket
173        sdp_socket = socket.socket(
174            family=socket.AF_INET6, type=socket.SOCK_DGRAM
175        )
176        # Get interface index
177        interface_index = socket.if_nametoindex(self.interface)
178        # Bind to the multicast group address
179        port = self.sdp_port
180
181        # Bind for IPv6 should use 4-tuple (host, port, flowinfo, scopeid)
182        # For AF_INET6 address family, a four-tuple (host, port, flowinfo, scope_id)
183        # is used, where flowinfo and scope_id represent the sin6_flowinfo
184        # and sin6_scope_id members in struct sockaddr_in6 in C
185        # From getaddrinfo() documentation:
186        # The function returns a list of 5-tuples with the following structure:
187        # (family, type, proto, canonname, sockaddr), interesting is sockaddr
188        # sockaddr is a tuple describing a socket address, whose format depends
189        # on the returned family (a (address, port) 2-tuple for AF_INET,
190        # a (address, port, flowinfo, scope_id) 4-tuple for AF_INET6),
191        # and is meant to be passed to the socket.connect() method or bind()
192        # The scope_id is a number that identifies the interface in a scope.
193        # can be obtained from if_nametoindex() or from the ip link show in linux
194        multicast_address = V2GTPAddress.MULTICAST_ADDRESS.value
195
196        logger.debug(
197            socket.getaddrinfo(
198                multicast_address,
199                port,
200                socket.AF_INET6,
201                socket.SOCK_DGRAM,
202                socket.SOL_UDP,
203            )
204        )
205        # Bind socket to multicast_address
206        sdp_socket.bind((multicast_address, port, 0, interface_index))
207
208        logger.debug(sdp_socket.getsockname())
209
210        # Without this was not possible to listen on multicast
211        multicast_addr_bin = socket.inet_pton(
212            socket.AF_INET6, multicast_address
213        )
214        mreq = multicast_addr_bin + struct.pack("I", interface_index)
215        sdp_socket.setsockopt(
216            socket.IPPROTO_IPV6, socket.IPV6_JOIN_GROUP, mreq
217        )
218
219        logger.debug(
220            "Starting SDP server, interface: %s interface-index: %s",
221            self.interface,
222            socket.if_nametoindex(self.interface),
223        )
224
225        try:
226            while not self.udp_stop_flag.is_set():
227                print("SDP server is running in while loop")
228                data, addr = sdp_socket.recvfrom(1024)
229                # TODO: Add parser for SDP request
230                logger.debug(f"Received {len(data)} bytes from {addr}: {data}")
231                sdp_request = V2GTPMessage(data)
232                requested_security, _ = sdp_request.parse_v2gtp_sdp_request()
233
234                # Add SDP request to the EVEnumerator if TLS_CHECK mode is enabled
235                if self.ev_enumerator is not None:
236                    if EVEnumMode.TLS_CHECK in self.ev_enumerator.enum_modes:
237                        self.ev_enumerator.sdp_request = sdp_request
238
239                # True => station will use TLS
240                if self.tls_flag:
241                    if requested_security == V2GTPProtocols.TLS:
242                        self.accept_security = True
243                    elif requested_security == V2GTPProtocols.NO_TLS:
244                        self.accept_security = False
245                else:
246                    if requested_security == V2GTPProtocols.TLS:
247                        self.accept_security = False
248                    elif requested_security == V2GTPProtocols.NO_TLS:
249                        self.accept_security = True
250
251                # True => the station use same security as the EVCC requested
252                if self.accept_security:
253                    if requested_security == V2GTPProtocols.TLS:
254                        self.tls_flag = True
255                    elif requested_security == V2GTPProtocols.NO_TLS:
256                        self.tls_flag = False
257                else:
258                    if requested_security == V2GTPProtocols.TLS:
259                        self.tls_flag = False
260                    elif requested_security == V2GTPProtocols.NO_TLS:
261                        self.tls_flag = True
262
263                logger.debug("TLS flag is set to: %s", self.tls_flag)
264
265                # Create proper SDP response message
266                sdp_response = sdp_request.create_response(
267                    ipv6=self.ipv6_address,
268                    port=self.tcp_port,
269                    protocol=self.protocol,
270                    tls_flag=self.tls_flag,
271                )
272                response_message = sdp_response
273                # Following line just for IDE, because it cannot recognize
274                assert isinstance(response_message, bytes)
275                sdp_socket.sendto(response_message, addr)
276
277                if not self.tcp_continue_flag.is_set():
278                    try:
279                        # TODO: Add option, user can set timeout in seconds
280                        await asyncio.wait_for(self.tcp_server(), timeout=5)
281                        print("TCP server connection established")
282                        # Process TCP data without timeout
283                        await self.v2gtp_comm_handler()
284                    except asyncio.TimeoutError:
285                        print("TCP server connection timeout")
286        except KeyboardInterrupt:
287            print("Stopping SDP server by KeyboardInterrupt")
288        finally:
289            print("SDP server stopped")
290            sdp_socket.close()

293        """Run TCP server.
294
295        Wait for connection from EVCC.
296        After connection is established, stop the SDP server
297        and wait for V2G communication session
298
299        Based on the tls_flag, for communication is used TCP or TLS protocol.
300        """
301        logger.debug("TCP server started")
302
303        with socket.socket(
304            family=socket.AF_INET6, type=socket.SOCK_STREAM
305        ) as server_sock:
306            # Get interface index
307            interface_index = socket.if_nametoindex(self.interface)
308
309            port = self.tcp_port
310            link_local_address = self.ipv6_address
311
312            # Add that to sdp_server cause it will inform the EVCC on which
313            # port to connect to the TCP server
314
315            # Avoid bind() exception: OSError: [Errno 48] Address already in use
316            server_sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
317            server_sock.bind((link_local_address, port, 0, interface_index))
318            server_sock.listen()
319            # This will accept only one client
320            #  server doesn't accept new connections for each iteration of the loop
321            # if i want that i should use while True and accept() in the loop
322            server_sock.setblocking(False)
323
324            if self.tls_flag:
325                # await self.tls_server(server_sock)
326                context = ssl.SSLContext(ssl.PROTOCOL_TLS_SERVER)
327                # OR
328                # context = ssl.create_default_context(ssl.Purpose.CLIENT_AUTH)
329                # TODO change, just to test with switchEV EVCC implementation
330                context.load_cert_chain(
331                    certfile=self.certfile_path,
332                    keyfile=self.keyfile_path,
333                    password="12345",  # TODO: Add password
334                )
335                # context.set_ecdh_curve("prime256v1")
336                logger.debug(self.certfile_path)
337                logger.debug(self.keyfile_path)
338                # TODO: Add module for setting and checking used TLS version
339                # and USED cipher suites setting min and max tls version
340                context.minimum_version = self.tls_version_min
341                context.maximum_version = self.tls_version_max
342                logger.debug(
343                    "Minimum version of TLS: %s", context.minimum_version
344                )
345                logger.debug(
346                    "Maximum version of TLS: %s", context.maximum_version
347                )
348                logger.debug(
349                    "Supported cipher suites: %s", context.get_ciphers()
350                )
351
352                with context.wrap_socket(
353                    server_sock, server_side=True
354                ) as ssock:
355                    while not self.tcp_continue_flag.is_set():
356                        try:
357                            conn, addr = ssock.accept()
358                            print("TLS, Connected by: ", addr)
359                            # Retrieve and print the negotiated cipher suite
360                            negotiated_cipher = conn.cipher()
361                            print(
362                                f"Negotiated Cipher Suite: {negotiated_cipher}"
363                            )
364                            # Retrieve and print the negotiated TLS version
365                            negotiated_tls_version = conn.version()
366                            print(
367                                f"Negotiated TLS Version: {negotiated_tls_version}"
368                            )
369                            # Retrieve shared cipher suites
370                            shared_ciphers = conn.shared_ciphers()
371                            if self.ev_enumerator is not None:
372                                if (
373                                    EVEnumMode.TLS_ENUM
374                                    in self.ev_enumerator.enum_modes
375                                ):
376                                    self.ev_enumerator.tls_version = (
377                                        negotiated_tls_version
378                                    )
379                                    self.ev_enumerator.cipher_suite = (
380                                        negotiated_cipher
381                                    )
382                                    self.ev_enumerator.shared_ciphers = (
383                                        shared_ciphers
384                                    )
385
386                            self.tcp_continue_flag.set()
387                            self.tcp_connection = conn
388                        except BlockingIOError:
389                            # No incoming connection, perform other tasks or wait
390                            await asyncio.sleep(0.1)
391            else:
392                while not self.tcp_continue_flag.is_set():
393                    try:
394                        conn, addr = server_sock.accept()
395                        print("Plain TCP, Connected by: ", addr)
396                        self.tcp_continue_flag.set()

398                    except BlockingIOError:
399                        # No incoming connection, perform other tasks or wait

168        The SDP server is started on UDP (multicast) port 15118 (defined in ISO15118-2).
169        Should accepts UDP packets with a local-link IP multicast destination address
170        """
171        print("SDP server started")
172
173        # Create a UDP socket
174        sdp_socket = socket.socket(
175            family=socket.AF_INET6, type=socket.SOCK_DGRAM
176        )
177        # Get interface index
178        interface_index = socket.if_nametoindex(self.interface)
179        # Bind to the multicast group address
180        port = self.sdp_port
181
182        # Bind for IPv6 should use 4-tuple (host, port, flowinfo, scopeid)
183        # For AF_INET6 address family, a four-tuple (host, port, flowinfo, scope_id)
184        # is used, where flowinfo and scope_id represent the sin6_flowinfo
185        # and sin6_scope_id members in struct sockaddr_in6 in C
186        # From getaddrinfo() documentation:
187        # The function returns a list of 5-tuples with the following structure:
188        # (family, type, proto, canonname, sockaddr), interesting is sockaddr
189        # sockaddr is a tuple describing a socket address, whose format depends
190        # on the returned family (a (address, port) 2-tuple for AF_INET,
191        # a (address, port, flowinfo, scope_id) 4-tuple for AF_INET6),
192        # and is meant to be passed to the socket.connect() method or bind()
193        # The scope_id is a number that identifies the interface in a scope.
194        # can be obtained from if_nametoindex() or from the ip link show in linux
195        multicast_address = V2GTPAddress.MULTICAST_ADDRESS.value
196
197        logger.debug(
198            socket.getaddrinfo(
199                multicast_address,
200                port,
201                socket.AF_INET6,
202                socket.SOCK_DGRAM,
203                socket.SOL_UDP,
204            )
205        )
206        # Bind socket to multicast_address
207        sdp_socket.bind((multicast_address, port, 0, interface_index))
208
209        logger.debug(sdp_socket.getsockname())
210
211        # Without this was not possible to listen on multicast
212        multicast_addr_bin = socket.inet_pton(
213            socket.AF_INET6, multicast_address
214        )
215        mreq = multicast_addr_bin + struct.pack("I", interface_index)
216        sdp_socket.setsockopt(
217            socket.IPPROTO_IPV6, socket.IPV6_JOIN_GROUP, mreq
218        )

220        logger.debug(
221            "Starting SDP server, interface: %s interface-index: %s",

234
235                # Add SDP request to the EVEnumerator if TLS_CHECK mode is enabled
236                if self.ev_enumerator is not None:
237                    if EVEnumMode.TLS_CHECK in self.ev_enumerator.enum_modes:
238                        self.ev_enumerator.sdp_request = sdp_request
239
240                # True => station will use TLS
241                if self.tls_flag:
242                    if requested_security == V2GTPProtocols.TLS:
243                        self.accept_security = True
244                    elif requested_security == V2GTPProtocols.NO_TLS:

246                else:

261                    elif requested_security == V2GTPProtocols.NO_TLS:
262                        self.tls_flag = True
263
264                logger.debug("TLS flag is set to: %s", self.tls_flag)

266                # Create proper SDP response message
267                sdp_response = sdp_request.create_response(
268                    ipv6=self.ipv6_address,
269                    port=self.tcp_port,
270                    protocol=self.protocol,
271                    tls_flag=self.tls_flag,
272                )
273                response_message = sdp_response
274                # Following line just for IDE, because it cannot recognize
275                assert isinstance(response_message, bytes)
276                sdp_socket.sendto(response_message, addr)
277
278                if not self.tcp_continue_flag.is_set():

289        finally:
290            print("SDP server stopped")
291            sdp_socket.close()
